
genetable <- read.csv(file="output.txt", sep=";",  header = TRUE, row.names = 1) #Чтение таблицы с наличием/отсуствием генов в переменную

genes <- df2genind(genematrix, sep="\t", type="PA")  #Преобразование матрицы в формат genind
df <- read.table(file="phenotypes_2.txt", sep = '\t', header = TRUE, row.names = 1)  #Чтение файла с фенотипами в переменную
phen <- as.vector(unlist(df))  #Преобразование в вектор
out <- treeWAS(snps = genes,
               phen = phen,
               tree = "BIONJ")
